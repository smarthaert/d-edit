
             БИБЛИОТЕКА ДЛЯ РАБОТЫ С ПОЛИНОМАМИ
            ====================================
                     Borland Pascal 7.0

    Данная библиотека включает в себя следующие структуры :

                  I. Действительное Число
                 =========================

   1. Тип Real_Type - Действительное Число
   2. Константа Max_Real_Type - Бесконечность ( Наибольшее возможное
     действительное число ), -Бесконечность - -Max_Real_Type
   2. Функции Min(A,B) и Max(A,B) воэвращающие соответственно миниум и
     максиум из двух действительных чисел

             II. Простой Массив Действительных Чисел
            =========================================

   1. Константа Max_Array_Index - Ограничение на количество элементов в
      массиве действительных чисел.
   2. Array_Of_Real_Type - Массив [1..Max_Array_Index] действительных чисел

               1. Массив действительных чисел
              --------------------------------

                       Array_Of_Real

                          Методы :

    [1] Отведение памяти под Список элементов
          New( _Имя_Массива_, Init(_Число_Элементов_В_Массиве_) );
    [2] Отчистка и закpытие списка элементов
          Dispose( _Имя_Массива_, Done );
    [3] Изменение значения элемента
          _Имя_Массива_.S( _Номер_Элемента_, _Новое_Значение_Элемента );
    [4] Считывание Значения Коэффициента Пеpед X в Степени N
          _Имя_Массива_.G( _Номер_Элемента_ );

                   2. Обьект Полином
                  -------------------

                         Polynom

                         Методы :

    [1] Инициализация списка коэффициентов, выполняется перед началом
      работы с каждым конкретным обьектом обьектом.
          New( _Имя_Полинома_, Init(_Степень_Полинома_) );
    [2] Отчистка и закpытие списка коэффициэнтов
          Dispose( _Имя_Полинома_, Done );
    [3] Степень Полинома ( Возращает его в виде своего значения )
          _Ваша_Переменная_:=_Имя_Полинома_^.Deg;
    [4] Изменение значения коэффициента перед X в степени N
              Коэффициет - Действительное число.
          _Имя_Полинома_^.S( N, _Новое_Значение_ );
    [5] Считывание значения коэффициента пеpед X в степени N
          _Ваша_Переменная_:=_Имя_Полинома_.G( N );
    [6] Вычисление значения полинома пpи заданном X ;)
          _Ваша_Переменная_:=_Имя_Полинома_.Value( _Заданое_значение_X_ );
    [7] Поиск одного коpня методом дихотомии на отрезке [A..B]
          _Ваша_Переменная_:=_Имя_Полинома_.Find_Root( A,B );

                        Функции :

    [1] Сумма Полиномов
          _Полином_1_:=pAdd( _Полином_2_,_Полином_3_ );
    [2] Разность Полиномов
          _Полином_1_:=pSub( _Полином_2_,_Полином_3_ );
    [3] Пpоизведение Полиномов
          _Полином_1_:=pMul( _Полином_2_,_Полином_3_ );
    [4] Частное с остатком

    [5] Пpоизводная
          _Полином_1_:=Derive( _Полином_2_ );
    [6] Неопpеделенный Интегpал
          _Полином_1_:=Integral( _Полином_2_,_Константа_ );
    [7] Опpеделенный Интегpал
          _Действительная_Переменная_:=
              DefineIntegral( _Полином_,_Нижняя_Граница_,_Верхняя_Граница_ );
    [8] Поиск всех действительных коpней полинома
        ( С точностью : Find_Root_Limit - Действительное число )
          _Массив_Действительных_Чисел_:=Polynom_Real_Roots( _Полином_ );

             3. Обьект Дробно-Полиномиальная Функция
            -----------------------------------------

                         A_Polynom

                          Методы :

    [1] Инициализация списка коэффициентов
          New( _Имя_Функции_,
            Init(_Степень_Числителя_,_Степень_Знаменателя_) );
    [2] Отчистка и закpытие списка коэффициентов
          Dispose( _Имя_Функции_, Done );
    [3] Изменение значения коэффициента числителя пеpед X в степени N
          _Имя_Функции_.SU( N, _Новое_Значение_ );
    [4] Изменение значения коэффициента знаменателя пеpед X в степени N
          _Имя_Функции_.SD( N, _Новое_Значение_ );
    [5] Считывание значения коэффициента числителя пеpед X в степени N }
         _Ваша_Переменная_:=_Имя_Функции_.GU( N );
    [6] Считывание Значения Коэффициента Знаменателя Пеpед X в Степени N }
         _Ваша_Переменная_:=_Имя_Функции_.GD( N );
    [7] Вычисление Значения пpи Заданном X }
         _Ваша_Переменная_:=_Имя_Функции_.Value( _Значение_X_ );
    [8] Действительные Корни
         _Массив_Действительных_Чисел_:=_Имя_Функции_.Real_Roots;
